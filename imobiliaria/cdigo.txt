use imobiliaria

CREATE TABLE proprietario (
IDproprietário int(11) PRIMARY KEY auto_increment,
nascimento date notnull,
rua varchar(255),
cidade varchar(255),
numerocasa varchar(10),
numerocelular varchar(15) notnull,
CPF varchar(15) unique notnull,
país varchar(20),
bairro varchar(255),
RG varchar(15) unique notnull,
nome varchar(255) notnull,
estado varchar(255),
cep varchar(255),
email varchar(255) unique not null,
contabancaria varchar(255) not null,
pix varchar(255) not null,
banco varchar(255) not null,
agencia varchar(255) not null
)


CREATE TABLE fiador (
IDfiador int(11) PRIMARY KEY auto_increment,
nascimento date not null,
nome varchar(50) not null,
CPF varchar(15) unique not null,
telefone varchar(15) not null,
email varchar(255) unique not null,
RG varchar(15) unique not null
)


CREATE TABLE corretor (
IDcorretor int(11) PRIMARY KEY auto_increment,
nascimento date not null,
RG varchar(20) unique not null,
telefone varchar(15) not null,
email varchar(255) unique not null,
nome varchar(255) not null,
CPF varchar(15) unique not null,
cidade varchar(255),
bairro varchar(255),
rua varchar(255),
numerocasa varchar(20),
complemento varcar(255)
)


CREATE TABLE inquilino (
IDinquilino int(11) PRIMARY KEY auto_increment,
nascimento date not null,
CPF varchar(15) unique not null,
email varchar(50) unique not null,
RG varchar(15) unique not null,
nome varchar(50) not null,
telefone varchar(15) not null,
IDfiador varchar(15),
FOREIGN KEY(IDfiador) REFERENCES fiador (IDfiador)
)


CREATE TABLE imovel (
IDimovel int(11) PRIMARY KEY auto_increment,
bairro varchar(50) not null,
estado varchar(50) not null,
cidade varchar(50) not null,
rua varchar(50) not null,
complemento varchar(50),
areatotal int(10,4) not null,
pais varchar(50) not null,
suite int(12) not null,
garagem bit not null,
numerocasa varchar(10) not null,
alugado bit not null,
banheiro int not null,
quarto int not null,
vagasgaragem int not null,
quintal bit not null,
lavanderia bit not null,
piscina bit not null,
areaconstruida int(10,4) not null,
ano int(2000) not null,
IDproprietário varchar(10),
FOREIGN KEY(IDproprietário) REFERENCES proprietario (IDproprietário)
)



CREATE TABLE aluguel (
IDaluguel int(11) PRIMARY KEY auto_increment,
datafim date not null,
valor decimal (4,2) not null,
datainicio date not null,
tipocontrato varchar(50) not null,
datavencimento varhar(10) not null,
descri varchar(500),
IDinquilino varchar(10),
IDimovel varchar(10),
IDcorretor varchar(10),
FOREIGN KEY(IDimovel) REFERENCES imovel (IDimovel),
FOREIGN KEY(IDinquilino) REFERENCES inquilino (IDinquilino),
FOREIGN KEY(IDcorretor) REFERENCES corretor (IDcorretor)
)


CREATE TABLE midias (
IDmidia int(11) PRIMARY KEY auto_increment,
foto varchar(255),
video varchar(255),
IDimovel varchar(10),
FOREIGN KEY(IDimovel) REFERENCES imovel (IDimovel)
)
