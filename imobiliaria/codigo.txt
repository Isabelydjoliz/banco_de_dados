-- Geração de Modelo físico
-- Sql ANSI 2003 - brModelo.

CREATE TABLE proprietario (
IDproprietário varchar(10) PRIMARY KEY,
nascimento date,
rua varchar(255),
cidade vachar(255),
numerocasa varchar(10),
numerocelular varchar(15),
CPF varchar(15) unique,
país varchar(20),
bairro varchar(255),
RG varchar(15),
nome varchar(255),
estado varchar(255),
cep varchar(255),
email varchar(255) unique,
contabancaria varchar(255),
pix varchar(255),
banco varchar(255),
agencia varchar(255)
)


CREATE TABLE fiador (
IDfiador varchar(15) PRIMARY KEY,
nascimento date,
nome varchar(50),
CPF varchar(15) unique,
telefone varchar(15),
email varchar(255),
RG varchar(15),
)


CREATE TABLE corretor (
IDcorretor varchar(10) PRIMARY KEY,
nascimento date,
RG varchar(20),
telefone varchar(15),
email varchar(255) unique,
nome varchar(255),
CPF varchar(15) unique,
cidade varchar(255),
bairro varchar(255),
rua varchar(255),
numerocasa varchar(20),
complemento varcar(255)
)


CREATE TABLE inquilino (
IDinquilino varchar(10) PRIMARY KEY,
nascimento date,
CPF varchar(15) unique,
email varchar(50) unique,
RG varchar(15),
nome varchar(50),
telefone varchar(15),
IDfiador varchar(15),
FOREIGN KEY(IDfiador) REFERENCES fiador (IDfiador)
)


CREATE TABLE imovel (
bairro varchar(50),
estado varchar(50),
cidade varchar(50),
rua varchar(50),
complemento varchar(50),
areatotal int(10,4),
pais varchar(50),
suite int(12),
garagem bit,
numerocasa varchar(10),
alugado bit,
banheiro int,
quarto int,
IDimovel varchar(10) PRIMARY KEY,
vagasgaragem int,
quintal bit,
lavanderia bit,
piscina bit,
areaconstruida int(10,4),
ano int,
IDproprietário varchar(10),
FOREIGN KEY(IDproprietário) REFERENCES proprietario (IDproprietário)
)



CREATE TABLE aluguel (
datafim date,
IDaluguel varchar(10) PRIMARY KEY,
valor decimal (4,2),
datainicio date,
tipocontrato varchar(50),
datavencimento varhar(10),
descri varchar(500),
IDinquilino varchar(10),
IDimovel varchar(10),
IDcorretor varchar(10),
FOREIGN KEY(IDimovel) REFERENCES imovel (IDimovel),
FOREIGN KEY(IDinquilino) REFERENCES inquilino (IDinquilino),
FOREIGN KEY(IDcorretor) REFERENCES corretor (IDcorretor)
)


CREATE TABLE midias (
IDmidia Texto(1) PRIMARY KEY,
foto varchar(255),
Atributo_1 varchar(255),
IDimovel varchar(10),
FOREIGN KEY(IDimovel) REFERENCES imovel (IDimovel)
)
